        $('#shopify-section-template--17138240979127__hp_one_daily_serving_LQkwzy .section_close').on('click', function() {

            $('#shopify-section-template--17138240979127__hp_one_daily_serving_LQkwzy .search_input input').val($(this).text());
            perform_search($(this).text());
        });
        $(document).on('click', function() {
            if (!$(event.target).closest('#shopify-section-template--17138240979127__hp_one_daily_serving_LQkwzy .section_holder').length) {
                $('#shopify-section-template--17138240979127__hp_one_daily_serving_LQkwzy .section_holder').removeClass('active');
            }
        });
        const processChange = debounce((event) => perform_search($('#shopify-section-template--17138240979127__hp_one_daily_serving_LQkwzy .search_input input').val()));

        function perform_search(search_text) {
            if (search_text != '') {
                $('#shopify-section-template--17138240979127__hp_one_daily_serving_LQkwzy .search_input input').val(search_text);
                $.ajax({
                    url: '/search?section_id=search_result_content&q=' + search_text + '&type=product',
                    type: 'GET'
                }).then(res => {
                    $('[ data-id="filled_search"]').show();
                    $('[ data-id="empty_search"]').hide();
                    $("#shopify-section-template--17138240979127__hp_one_daily_serving_LQkwzy .best_sellers_contents_wrapper[data-id='filled_search']").html($(res).children());
                });
                $.ajax({
                    url: '/search/suggest.json?q=' + search_text + '&resources[type]=collection',
                    type: 'GET'
                }).then(res => {
                    $.ajax(res.resources.results.collections[0].handle)
                });
            } else {
                $('[ data-id="filled_search"]').hide();
                $('[ data-id="empty_search"]').show();
            }
        }
    </script>
